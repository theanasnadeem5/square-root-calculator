        <option value="basic">Basic Method (√x)</option>
        <option value="factorization">Factorization Method</option>
        <option value="division">Long Division Method</option>
    </select>
    <button onclick="calculate()">Calculate</button>
    <div class="result" id="result"></div>
</div>

<script>
function calculate() {
    let num = document.getElementById('number').value;
    let method = document.getElementById('method').value;
    let resultArea = document.getElementById('result');

    if (num === "" || isNaN(num) || num < 0) {
        resultArea.innerHTML = "Please enter a valid non-negative number.";
        return;
    }

    num = Number(num);
    let result;

    if (method === 'basic') {
        result = Math.sqrt(num).toFixed(4);
        resultArea.innerHTML = `Square Root (Basic Method): ${result}`;
    } else if (method === 'factorization') {
        resultArea.innerHTML = `Factorization Method result: √${num} ≈ ${Math.sqrt(num).toFixed(4)} (detailed steps coming soon)`;
    } else if (method === 'division') {
        resultArea.innerHTML = `Long Division Method result: √${num} ≈ ${Math.sqrt(num).toFixed(4)} (detailed steps coming soon)`;
    }
}
</script>
